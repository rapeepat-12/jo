#!/usr/bin/env python3 """ Incremental Stats Game üßÆ A command-line incremental game designed to run on GitHub Codespaces or locally. """ import math import time import sys # ====== CORE GAME CLASS ====== class Game: def __init__(self): self.stats = {"a": 1.0} self.unlocked = ["a"] self.all_boost = 1 self.a_boost = 1 self.upgrade_price = 100 self.unlock_cost = 1e3 self.challenge_level = 0 self.challenge_active = False # ====== GROWTH LOOP ====== def tick(self): """Grow base stat over time.""" growth = self.stats["a"] * 0.1 * self.all_boost * self.a_boost self.stats["a"] += growth return self.stats["a"] # ====== UPGRADES ====== def buy_upgrade(self): """Buy x2 global boost upgrade.""" if self.stats["a"] >= self.upgrade_price: self.stats["a"] -= self.upgrade_price self.all_boost *= 2 self.upgrade_price *= 10 print("‚úÖ Upgrade bought! All boost √ó2.") else: print("‚ùå Not enough 'a'!") # ====== UNLOCK NEW STAT ====== def unlock_next(self): """Unlock next stat (b, c, d, etc.).""" if self.stats["a"] >= self.unlock_cost: self.stats["a"] -= self.unlock_cost next_stat = self._next_stat_name(self.unlocked[-1]) self.unlocked.append(next_stat) self.stats[next_stat] = 0.0 self.unlock_cost *= 10 print(f"üîì Unlocked new stat: {next_stat}") else: print("‚ùå Not enough 'a' to unlock next stat!") def _next_stat_name(self, name): """Generate next stat name (a ‚Üí b ‚Üí ... ‚Üí z ‚Üí A ‚Üí ... ‚Üí Z ‚Üí aa ‚Üí ab ...)""" alphabet = [chr(i) for i in range(97, 123)] + [chr(i) for i in range(65, 91)] if name not in alphabet: prefix, last = name[:-1], name[-1] if last == "Z": return prefix + "a" else: return prefix + alphabet[alphabet.index(last) + 1] else: if name == "Z": return "aa" return alphabet[alphabet.index(name) + 1] # ====== CHALLENGES ====== def start_challenge(self): """Start a new challenge ‚Äî resets progress but gives reward after completion.""" if self.challenge_active: print("‚ö†Ô∏è Already in a challenge!") return self.challenge_active = True self.challenge_level += 1 self._reset_progress() print(f"üèÅ Challenge {self.challenge_level} started! Your progress is reset to 'a'.") def complete_challenge(self): """Complete challenge and apply reward.""" if not self.challenge_active: print("‚ùå You are not in a challenge.") return reward = 1 + (self.challenge_level * 0.5) self.a_boost *= reward self.challenge_active = False print(f"üéâ Challenge {self.challenge_level} completed! 'a' boost √ó{reward:.2f}") def _reset_progress(self): """Reset all stats up to aA when entering challenge.""" self.stats = {"a": 1.0} self.unlocked = ["a"] self.all_boost = 1 self.a_boost = 1 self.upgrade_price = 100 self.unlock_cost = 1e3 # ====== INFO DISPLAY ====== def show(self): print("\nüìä --- GAME STATUS ---") print(f"Stats: {self.stats}") print(f"Unlocked: {self.unlocked}") print(f"All boost: √ó{self.all_boost}") print(f"A boost: √ó{self.a_boost}") print(f"Upgrade cost: {self._format(self.upgrade_price)} a") print(f"Next unlock cost: {self._format(self.unlock_cost)} a") print(f"Challenge: {'Active' if self.challenge_active else 'None'} (Level {self.challenge_level})") print("-----------------------\n") def _format(self, n): """Human-readable number format with suffixes (K, M, B, etc.).""" suffixes = ["", "K", "M", "B", "T", "Qd", "Qn", "Sx", "Sp", "Oc", "No", "De"] if n < 1000: return f"{n:.2f}" exp = int(math.log10(n) // 3) if exp >= len(suffixes): exp = len(suffixes) - 1 return f"{n / (10 ** (3 * exp)):.2f}{suffixes[exp]}" # ====== MAIN GAME LOOP ====== def main(): game = Game() print("üéÆ Welcome to the Incremental Stats Game!") print("Commands: tick / upgrade / unlock / challenge / complete / show / exit") while True: cmd = input("> ").strip().lower() if cmd == "tick": a = game.tick() print(f"Tick... a = {game._format(a)}") elif cmd == "upgrade": game.buy_upgrade() elif cmd == "unlock": game.unlock_next() elif cmd == "challenge": game.start_challenge() elif cmd == "complete": game.complete_challenge() elif cmd == "show": game.show() elif cmd == "exit": print("üëã Goodbye!") break else: print("‚ùì Unknown command!") if __name__ == "__main__": main()
